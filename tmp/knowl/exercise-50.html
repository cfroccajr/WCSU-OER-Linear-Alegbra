\begin{equation*} C=\left[ \begin{array}{rrrr} 1 \amp a \amp a^2 \amp a^3\\ 1 \amp b \amp b^2 \amp b^3\\ 1 \amp c \amp c^2 \amp c^3\\ 1 \amp d \amp d^2 \amp d^3\\ \end{array} \right]. \end{equation*}
